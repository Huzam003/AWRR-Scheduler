1. struct sched_class - COMPLETE FUNCTION POINTER LIST
================================================================================

File: kernel/sched/sched.h

struct sched_class {

#ifdef CONFIG_UCLAMP_TASK
    int uclamp_enabled;
#endif

    /* --- Core scheduling operations --- */

    void (*enqueue_task)(struct rq *rq, struct task_struct *p, int flags);
        // Called when a task becomes runnable (wakeup, fork, migration).
        // flags: ENQUEUE_WAKEUP, ENQUEUE_RESTORE, ENQUEUE_MIGRATED, etc.

    void (*dequeue_task)(struct rq *rq, struct task_struct *p, int flags);
        // Called when a task leaves runnable state (sleep, exit, migration).
        // flags: DEQUEUE_SLEEP, DEQUEUE_SAVE, DEQUEUE_MIGRATING, etc.

    void (*yield_task)(struct rq *rq);
        // Called when a task voluntarily yields the CPU (sched_yield()).

    bool (*yield_to_task)(struct rq *rq, struct task_struct *p);
        // Yield to a specific task. Returns true if successful.

    void (*check_preempt_curr)(struct rq *rq, struct task_struct *p, int flags);
        // Check if newly woken task 'p' should preempt currently running task.
        // Called after enqueue on wakeup. Sets TIF_NEED_RESCHED if needed.

    struct task_struct *(*pick_next_task)(struct rq *rq);
        // Select the next task to run from this class's runqueue.
        // Returns NULL if no runnable task exists in this class.

    void (*put_prev_task)(struct rq *rq, struct task_struct *p);
        // Called when task 'p' is about to be switched out of the CPU.
        // Update accounting, re-insert into class data structures.

    void (*set_next_task)(struct rq *rq, struct task_struct *p, bool first);
        // Called when task 'p' is about to become the current running task.
        // 'first' is true on initial pick, false on subsequent updates.

#ifdef CONFIG_SMP
    /* --- SMP / Load Balancing operations --- */

    int (*balance)(struct rq *rq, struct task_struct *prev, struct rq_flags *rf);
        // Called from pick_next_task context. Opportunity to pull tasks.
        // Returns 1 if a task was pulled and balance was performed.

    int (*select_task_rq)(struct task_struct *p, int task_cpu, int flags);
        // Select which CPU a task should run on.
        // Called during wake-up and fork to determine target CPU.
        // Returns the selected CPU number.

    struct task_struct *(*pick_task)(struct rq *rq);
        // Pick a task from this class for core scheduling (CONFIG_SCHED_CORE).
        // Does NOT remove the task from runqueue. Read-only pick.

    void (*migrate_task_rq)(struct task_struct *p, int new_cpu);
        // Called when a task is being migrated to new_cpu.

    void (*task_woken)(struct rq *this_rq, struct task_struct *task);
        // Called after a task is woken on this rq.

    void (*set_cpus_allowed)(struct task_struct *p,
                             const struct cpumask *newmask, u32 flags);
        // Called when a task's CPU affinity mask is changed.

    void (*rq_online)(struct rq *rq);
        // Called when a CPU comes online.

    void (*rq_offline)(struct rq *rq);
        // Called when a CPU goes offline.

    struct rq *(*find_lock_rq)(struct task_struct *p, struct rq *rq);
        // Find and lock the runqueue containing task p.
        // Used for task pushing/pulling in RT/DL.
#endif

    /* --- Timer and lifecycle callbacks --- */

    void (*task_tick)(struct rq *rq, struct task_struct *p, int queued);
        // Called from scheduler_tick() on each timer interrupt.
        // 'queued' indicates if this task is on the runqueue.
        // Opportunity to update time accounting and check preemption.

    void (*task_fork)(struct task_struct *p);
        // Called when a new child task is forked.
        // Initialize scheduling entity for the child.

    void (*task_dead)(struct task_struct *p);
        // Called when a task exits (in do_exit).

    /* --- Policy switching callbacks --- */

    void (*switched_from)(struct rq *this_rq, struct task_struct *task);
        // Called when a task switches AWAY from this scheduling class.

    void (*switched_to)(struct rq *this_rq, struct task_struct *task);
        // Called when a task switches TO this scheduling class.

    void (*prio_changed)(struct rq *this_rq, struct task_struct *task,
                         int oldprio);
        // Called when a task's priority changes while in this class.

    unsigned int (*get_rr_interval)(struct rq *rq, struct task_struct *task);
        // Return the time slice for sched_rr_get_interval() syscall.
        // Return value in jiffies.

    void (*update_curr)(struct rq *rq);
        // Update current task's runtime statistics.
        // Called from various places to keep accounting current.

#ifdef CONFIG_FAIR_GROUP_SCHED
    void (*task_change_group)(struct task_struct *p);
        // Called when a task changes cgroup.
#endif
};

2. SCHEDULING CLASS HIERARCHY AND ORDERING
================================================================================

File: kernel/sched/sched.h
File: include/asm-generic/vmlinux.lds.h

--- Hierarchy (highest to lowest priority) ---

    stop_sched_class        (highest - migration/stop tasks, 1 per CPU)
        |
    dl_sched_class          (SCHED_DEADLINE tasks)
        |
    rt_sched_class          (SCHED_FIFO, SCHED_RR tasks)
        |
    *** AWRR GOES HERE ***  (insert between rt and fair)
        |
    fair_sched_class        (SCHED_NORMAL, SCHED_BATCH tasks - CFS)
        |
    idle_sched_class        (lowest - idle task, 1 per CPU)


--- Modern Mechanism (kernel 5.9+ including 6.1): Linker Section Ordering ---

In kernel 6.1, the old linked-list 'next' pointer is GONE. Instead, scheduling
classes are ordered using linker sections and iterated using pointer arithmetic.

The DEFINE_SCHED_CLASS macro (kernel/sched/sched.h):

    #define DEFINE_SCHED_CLASS(name) \
        const struct sched_class name##_sched_class \
            __aligned(__alignof__(struct sched_class)) \
            __section("__" #name "_sched_class")

This places each sched_class instance into a named ELF section like:
    "__stop_sched_class"
    "__dl_sched_class"
    "__rt_sched_class"
    "__fair_sched_class"
    "__idle_sched_class"

The linker script (include/asm-generic/vmlinux.lds.h) orders them:

    #define SCHED_DATA                  \
        STRUCT_ALIGN();                 \
        __sched_class_highest = .;      \
        *(__stop_sched_class)           \
        *(__dl_sched_class)             \
        *(__rt_sched_class)             \
        *(__fair_sched_class)           \
        *(__idle_sched_class)           \
        __sched_class_lowest = .;

Boundary symbols:
    extern struct sched_class __sched_class_highest[];
    extern struct sched_class __sched_class_lowest[];

Iteration macros (kernel/sched/sched.h):

    #define for_class_range(class, _from, _to) \
        for (class = (_from); class < (_to); class++)

    #define for_each_class(class) \
        for_class_range(class, __sched_class_highest, __sched_class_lowest)

NOTE: Classes are laid out in memory from HIGHEST to LOWEST priority.
The loop iterates from highest priority to lowest. The first class
whose pick_next_task() returns non-NULL wins.

Extern declarations for all classes:
    extern const struct sched_class stop_sched_class;
    extern const struct sched_class dl_sched_class;
    extern const struct sched_class rt_sched_class;
    extern const struct sched_class fair_sched_class;
    extern const struct sched_class idle_sched_class;


--- HOW TO INSERT AWRR BETWEEN RT AND FAIR ---

1. In kernel/sched/awrr.c, define:
       DEFINE_SCHED_CLASS(awrr) = { ... };
   This creates section "__awrr_sched_class".

2. In include/asm-generic/vmlinux.lds.h, modify SCHED_DATA:
       #define SCHED_DATA                  \
           STRUCT_ALIGN();                 \
           __sched_class_highest = .;      \
           *(__stop_sched_class)           \
           *(__dl_sched_class)             \
           *(__rt_sched_class)             \
           *(__awrr_sched_class)           \   <-- ADD THIS LINE
           *(__fair_sched_class)           \
           *(__idle_sched_class)           \
           __sched_class_lowest = .;

3. In kernel/sched/sched.h, add extern:
       extern const struct sched_class awrr_sched_class;

4. The for_each_class macro will automatically iterate through AWRR
   in the correct priority position.

        return policy == SCHED_AWRR;

        return policy == SCHED_NORMAL || policy == SCHED_BATCH ||
               policy == SCHED_IDLE || policy == SCHED_FIFO ||
               policy == SCHED_RR || policy == SCHED_DEADLINE ||

    }

6. core.c - SCHEDULER REGISTRATION AND __schedule() LOOP
================================================================================

File: kernel/sched/core.c

--- Per-CPU runqueue declaration ---

    DEFINE_PER_CPU_SHARED_ALIGNED(struct rq, runqueues);

--- The main scheduling function ---

The __schedule() function is the heart of the Linux scheduler:

    static void __sched notrace __schedule(unsigned int sched_mode)
    {
        struct task_struct *prev, *next;
        unsigned long *switch_count;
        unsigned long prev_state;
        struct rq_flags rf;
        struct rq *rq;
        int cpu;

        cpu = smp_processor_id();
        rq = cpu_rq(cpu);
        prev = rq->curr;

        /* ... handle task state transitions, dequeue sleeping tasks ... */

        /* If prev is sleeping/dead, dequeue it */
        if (!preempt && prev_state) {
            if (signal_pending_state(prev_state, prev)) {
                WRITE_ONCE(prev->__state, TASK_RUNNING);
            } else {
                deactivate_task(rq, prev, DEQUEUE_SLEEP | DEQUEUE_NOCLOCK);
                /* ... */
            }
        }

        /* Pick the next task - THIS IS THE KEY PART */
        next = pick_next_task(rq, prev, &rf);

        /* Context switch if different task selected */
        if (likely(prev != next)) {
            rq->nr_switches++;
            rq->curr = next;
            /* ... update stats ... */
            rq = context_switch(rq, prev, next, &rf);
        } else {
            rq_unpin_lock(rq, &rf);
            __balance_callbacks(rq);
            raw_spin_rq_unlock_irq(rq);
        }
    }

--- pick_next_task() ---

There are two versions: an optimized one and a generic one.

Optimized path (when all tasks are CFS):

    static inline struct task_struct *
    __pick_next_task(struct rq *rq, struct task_struct *prev, struct rq_flags *rf)
    {
        const struct sched_class *class;
        struct task_struct *p;

        /* Optimization: if all tasks are in fair class, go straight to CFS */
        if (likely(!sched_class_above(prev->sched_class, &fair_sched_class) &&
                   rq->nr_running == rq->cfs.h_nr_running)) {

            p = pick_next_task_fair(rq, prev, rf);
            if (unlikely(p == RETRY_TASK))
                goto restart;
            if (!p) {
                put_prev_task(rq, prev);
                p = pick_next_task_idle(rq);
            }
            return p;
        }

    restart:
        put_prev_task_balance(rq, prev, rf);

        /* Generic path: iterate through all classes highest to lowest */
        for_each_class(class) {
            p = class->pick_next_task(rq);
            if (p)
                return p;
        }
        /* Should never reach here - idle class always returns a task */
        BUG();
    }

NOTE: The for_each_class(class) macro iterates from __sched_class_highest
to __sched_class_lowest. When AWRR is inserted between RT and FAIR in the
linker script, it will be checked after RT and before FAIR automatically.

--- schedule() wrapper ---

    asmlinkage __visible void __sched schedule(void)
    {
        struct task_struct *tsk = current;

        sched_submit_work(tsk);
        do {
            preempt_disable();
            __schedule(SM_NONE);
            sched_preempt_enable_no_resched();
        } while (need_resched());
        sched_update_worker(tsk);
    }
    EXPORT_SYMBOL(schedule);

--- Helper functions used in the scheduling path ---

    static inline void enqueue_task(struct rq *rq, struct task_struct *p,
                                    int flags)
    {
        /* ... update stats ... */
        p->sched_class->enqueue_task(rq, p, flags);
    }

    static inline void dequeue_task(struct rq *rq, struct task_struct *p,
                                    int flags)
    {
        /* ... update stats ... */
        p->sched_class->dequeue_task(rq, p, flags);
    }

    static inline void put_prev_task(struct rq *rq, struct task_struct *p)
    {
        WARN_ON_ONCE(rq->curr != p);
        p->sched_class->put_prev_task(rq, p);
    }

    static inline void set_next_task(struct rq *rq, struct task_struct *p)
    {
        p->sched_class->set_next_task(rq, p, false);
    }

--- Scheduling class priority comparison ---

    static inline bool sched_class_above(const struct sched_class *a,
                                         const struct sched_class *b)
    {
        /* Classes are ordered highest-to-lowest in memory.
         * Higher priority = lower address. */
        return a < b;
    }

8. FIXED-POINT ARITHMETIC IN THE KERNEL
================================================================================

File: kernel/sched/sched.h (load_weight, WMULT_*)
File: kernel/sched/core.c (weight arrays, set_load_weight)
File: kernel/sched/fair.c (__calc_delta, calc_delta_fair)

The kernel CANNOT use floating-point arithmetic. All scheduler math uses
integer arithmetic with fixed-point techniques.

--- Load Weight Structure ---

    struct load_weight {
        unsigned long   weight;       // Task weight
        u32             inv_weight;   // Precomputed inverse: (2^32) / weight
    };

--- Weight Constants ---

    #define NICE_0_LOAD_SHIFT  10
    #define NICE_0_LOAD        (1L << NICE_0_LOAD_SHIFT)  /* = 1024 */

    #define WMULT_SHIFT        32    /* Fixed-point shift for inverse weights */
    #define WMULT_CONST        (~0UL) /* = 2^32 - 1 = 4294967295 */

    #define WEIGHT_IDLEPRIO    3
    #define WMULT_IDLEPRIO     1431655765  /* (2^32) / 3 */

--- set_load_weight() ---

    static void set_load_weight(struct task_struct *p, bool update_load)
    {
        int prio = p->static_prio - MAX_RT_PRIO;  /* 0..39 from nice */
        struct load_weight *load = &p->se.load;

        if (task_has_idle_policy(p)) {
            load->weight = scale_load(WEIGHT_IDLEPRIO);
            load->inv_weight = WMULT_IDLEPRIO;
            return;
        }

        if (update_load && p->sched_class == &fair_sched_class) {
            reweight_task(p, prio);
        } else {
            load->weight = scale_load(sched_prio_to_weight[prio]);
            load->inv_weight = sched_prio_to_wmult[prio];
        }
    }

--- Fixed-Point Division via Multiplication ---

The key trick: instead of dividing by weight, multiply by inv_weight
and right-shift by WMULT_SHIFT (32).

    delta_weighted = (delta_exec * NICE_0_LOAD * inv_weight) >> WMULT_SHIFT

This is equivalent to:
    delta_weighted = delta_exec * (NICE_0_LOAD / weight)

But avoids the division entirely!

    static u64 __calc_delta(u64 delta_exec, unsigned long weight,
                            struct load_weight *lw)
    {
        u64 fact = scale_load_down(weight);
        int shift = WMULT_SHIFT;

        __update_inv_weight(lw);

        if (unlikely(fact >> 32)) {
            while (fact >> 32) {
                fact >>= 1;
                shift--;
            }
        }

        fact = (u64)(u32)fact * lw->inv_weight;

        while (fact >> 32) {
            fact >>= 1;
            shift--;
        }

        return mul_u64_u32_shr(delta_exec, fact, shift);
    }

--- calc_delta_fair() - CFS vruntime calculation ---

    static inline u64 calc_delta_fair(u64 delta, struct sched_entity *se)
    {
        if (unlikely(se->load.weight != NICE_0_LOAD))
            delta = __calc_delta(delta, NICE_0_LOAD, &se->load);
        return delta;
    }

    // This computes:  vruntime_delta = wall_time * (1024 / task_weight)
    // A task with weight 1024 (nice 0) gets vruntime == wall time.
    // A task with weight 2048 (nice -5ish) gets vruntime = wall_time / 2.
    // A task with weight 512  (nice +5ish) gets vruntime = wall_time * 2.


--- FIXED-POINT PATTERNS FOR AWRR ---

For AWRR, you can use the same pattern:

    /* AWRR fixed-point weight representation */
    #define AWRR_WEIGHT_SHIFT   10
    #define AWRR_WEIGHT_BASE    (1 << AWRR_WEIGHT_SHIFT)  /* 1024 */

    /* Time slice calculation (avoid float):
     * slice = base_slice * weight / total_weight
     * Using fixed-point:
     * slice = (base_slice * weight + total_weight/2) / total_weight
     * Or with shift:
     * slice = (base_slice * weight * inv_total) >> SHIFT
     */

    static inline unsigned int awrr_calc_slice(unsigned int base_slice,
                                               unsigned int weight,
                                               unsigned int total_weight)
    {
        u64 slice;
        if (unlikely(!total_weight))
            return base_slice;
        slice = (u64)base_slice * weight;
        /* Round to nearest */
        slice = (slice + total_weight / 2) / total_weight;
        /* Ensure minimum of 1 tick */
        return max_t(unsigned int, (unsigned int)slice, 1);
    }

--- update_curr() - The core accounting function ---

    static void update_curr(struct cfs_rq *cfs_rq)
    {
        struct sched_entity *curr = cfs_rq->curr;
        u64 now = rq_clock_task(rq_of(cfs_rq));
        u64 delta_exec;

        if (unlikely(!curr))
            return;

        delta_exec = now - curr->exec_start;
        if (unlikely((s64)delta_exec <= 0))
            return;

        curr->exec_start = now;
        curr->sum_exec_runtime += delta_exec;

        /* Update vruntime (weighted by inverse of task weight) */
        curr->vruntime += calc_delta_fair(delta_exec, curr);

        /* Update min_vruntime */
        update_min_vruntime(cfs_rq);

        /* ... account to cgroup, update bandwidth ... */
    }
